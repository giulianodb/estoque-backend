--Cria tabela temporaria

CREATE TABLE estoque.tmp ( id_menor int4 NULL, cpf varchar NULL );

update estoque.pessoa set cpf = null where cpf = '00000000000' or cpf='000000000000000' or cpf = '00000000000' or cpf = '0';

--Inserir qual é o menor id de cada Cpf duplicado

insert into estoque.tmp (cpf,id_menor) select distinct f.cpf , min(f.id) as menorId from estoque.pessoa f inner join estoque.pessoa f2 on f.cpf = f2.cpf where f.id <> f2.id group by f.cpf;

--Criar coluna cpf na tabela lte

ALTER TABLE estoque.lotemovimentacao ADD cpf_tmp varchar NULL;

--Adicionar o CPF da familia no campo temporaria

update estoque.lotemovimentacao set cpf_tmp = fa.cpf from (select cpf,id from estoque.pessoa ) as fa where fa.id = doador_id or fa.id = doadorcampanha_id ;

--Atualiza o id da família para o menor id com o mesmo CPF

update estoque.lotemovimentacao set doador_id = tmp.id_menor from (select cpf,id_menor from estoque.tmp ) as tmp where tmp.cpf = cpf_tmp and doador_id notnull ;

--Atualiza o id da família para o menor id com o mesmo CPF

update estoque.lotemovimentacao set doadorcampanha_id = tmp.id_menor from (select cpf,id_menor from estoque.tmp ) as tmp where tmp.cpf = cpf_tmp and doadorcampanha_id notnull ;

delete from estoque.pessoa where cpf in (select cpf from estoque.tmp) and id NOT IN (select id_menor from estoque.tmp);

DROP TABLE estoque.tmp;

ALTER TABLE estoque.lotemovimentacao DROP COLUMN cpf_tmp;

ALTER TABLE estoque.pessoa ADD CONSTRAINT cpf_un UNIQUE (cpf);

CREATE TABLE estoque.tmp ( id_menor int4 NULL, cpf varchar NULL );

update estoque.familia set cpfresponsavel = null where cpfresponsavel = '00000000000';

--Inserir qual é o menor id de cada Cpf duplicado

insert into estoque.tmp (cpf,id_menor)
select distinct f.cpfresponsavel , min(f.id) as menorId from estoque.familia f
inner join estoque.familia f2 on f.cpfresponsavel = f2.cpfresponsavel where f.id <> f2.id group by f.cpfresponsavel;

--Criar coluna cpf na tabela lte

ALTER TABLE estoque.lotemovimentacao ADD cpf_tmp varchar NULL;

--Adicionar o CPF da familia no campo temporaria

update estoque.lotemovimentacao set cpf_tmp = fa.cpfresponsavel from (select cpfresponsavel,id from estoque.familia ) as fa where fa.id = familia_id or fa.id = familiacampanha_id ;

--Atualiza o id da família para o menor id com o mesmo CPF

update estoque.lotemovimentacao set familia_id = tmp.id_menor from (select cpf,id_menor from estoque.tmp ) as tmp where tmp.cpf = cpf_tmp and familia_id notnull ;

--Atualiza o id da família para o menor id com o mesmo CPF

update estoque.lotemovimentacao set familiacampanha_id = tmp.id_menor from (select cpf,id_menor from estoque.tmp ) as tmp where tmp.cpf = cpf_tmp and familiacampanha_id notnull ;

delete from estoque.familia where cpfresponsavel in (select cpf from estoque.tmp) and id NOT IN (select id_menor from estoque.tmp);

DROP TABLE estoque.tmp;

ALTER TABLE estoque.lotemovimentacao DROP COLUMN cpf_tmp;
